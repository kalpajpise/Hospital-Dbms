CREATE TABLE EMPLOYEE (
E_ID INT(4) NOT NULL,
E_NAME VARCHAR(20) NOT NULL,
E_ADDRESS VARCHAR(20) NOT NULL,
E_PHONE INT(10),
E_JOB VARCHAR(20) NOT NULL,
E_SALARY INT(10)NOT NULL,
PRIMARY KEY (E_ID)
);

CREATE TABLE PATIENT(
P_ID INT(4)NOT NULL,
E_ID INT(4)NOT NULL,
P_NAME VARCHAR(20)NOT NULL,
P_ADDRESS VARCHAR (20)NOT NULL,
P_PHONE INT(10),
D_O_AD DATE ,
D_O_DIS DATE,
PRIMARY KEY(P_ID),
FOREIGN KEY(E_ID) REFERENCES EMPLOYEE(E_ID)ON DELETE CASCADE 
);

CREATE TABLE DOCTOR(
E_ID INT(4)NOT NULL,
P_ID INT(4)NOT NULL,
D_NAME VARCHAR(20)NOT NULL,
D_SPECS VARCHAR(20)NOT NULL,
FOREIGN KEY(E_ID) REFERENCES EMPLOYEE(E_ID)ON DELETE CASCADE, 
FOREIGN KEY(P_ID) REFERENCES PATIENT(P_ID)ON DELETE CASCADE 
);

CREATE TABLE NURSE(
E_ID INT(4)NOT NULL,
P_ID INT(4)NOT NULL,
N_NAME VARCHAR(20)NOT NULL,
FOREIGN KEY(E_ID) REFERENCES EMPLOYEE(E_ID)ON DELETE CASCADE, 
FOREIGN KEY(P_ID) REFERENCES PATIENT(P_ID)ON DELETE CASCADE 
);


CREATE TABLE RECEPTION(
E_ID INT(4)NOT NULL,
R_NAME VARCHAR(20)NOT NULL,
FOREIGN KEY(E_ID) REFERENCES EMPLOYEE(E_ID)ON DELETE CASCADE
);


CREATE TABLE RECORDS(
R_NO INT(4)NOT NULL,
P_ID INT(4)NOT NULL,
E_ID INT(4)NOT NULL,
//E_TYPE VARCHAR(20)NOT NULL,
R_DETAIL VARCHAR(30)NOT NULL,
PRIMARY KEY(R_NO),
FOREIGN KEY(E_ID) REFERENCES EMPLOYEE(E_ID)ON DELETE CASCADE, 
FOREIGN KEY(P_ID) REFERENCES PATIENT(P_ID)ON DELETE CASCADE 
);


CREATE TABLE ROOMS(
RE_ID INT(4) NOT NULL,
P_ID INT(4) NOT NULL,
R_TYPE VARCHAR(20) NOT NULL,
E_ID INT(4)NOT NULL,
E_TYPE VARCHAR(20)NOT NULL,
R_NO
